## 🛠️ AUTOMATION

### Video Gen Script
```bash
#!/bin/bash
# Generate videos across platforms

PLATFORMS=("runway" "pika" "kling" "haiper")
PROMPTS_DIR="./video-prompts"
OUTPUT_DIR="./videos"

for platform in "${PLATFORMS[@]}"; do
  for prompt in "$PROMPTS_DIR"/*.txt; do
    echo "Generating with $platform: $prompt"
    # Platform-specific API call
    generate_video "$platform" "$prompt" "$OUTPUT_DIR"
  done
done
```
